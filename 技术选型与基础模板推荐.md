## 🔧 核心模块设计

### 1. 剧本生成模块 (Script Service)

```go
type ScriptService struct {
    llmClient    *openai.Client
    promptTmpl   *template.Template
}

func (s *ScriptService) GenerateScript(ctx context.Context, req GenerateScriptRequest) (*Script, error) {
    // 1. 生成故事大纲
    outline, err := s.generateOutline(ctx, req.Theme)
    
    // 2. 生成分镜脚本
    scenes, err := s.generateScenes(ctx, outline)
    
    // 3. 提取角色信息
    characters, err := s.extractCharacters(ctx, scenes)
    
    // 4. 生成对话
    dialogues, err := s.generateDialogues(ctx, scenes)
    
    return &Script{
        Outline:    outline,
        Scenes:     scenes,
        Characters: characters,
        Dialogues:  dialogues,
    }, nil
}
```

### 2. 角色管理模块 (Character Service)

```go
type CharacterService struct {
    repo         repository.CharacterRepository
    comfyClient  *comfyui.Client
    ipadapter    *ipadapter.Client
}

func (s *CharacterService) GenerateCharacter(ctx context.Context, req GenerateCharacterRequest) (*Character, error) {
    // 1. 生成角色描述
    description := s.generateDescription(ctx, req)
    
    // 2. 生成参考图
    referenceImages, err := s.generateReferenceImages(ctx, description)
    
    // 3. 训练LoRA (可选)
    if req.TrainLoRA {
        loraModel, err := s.trainLoRA(ctx, referenceImages)
    }
    
    // 4. 生成三视图
    views, err := s.generateThreeViews(ctx, referenceImages)
    
    return &Character{
        Description: description,
        References:  referenceImages,
        LoRAModel:   loraModel,
        Views:       views,
    }, nil
}

func (s *CharacterService) ApplyConsistency(ctx context.Context, character *Character, imageURL string) (string, error) {
    // 使用IPAdapter保持角色一致性
    return s.ipadapter.Generate(ctx, imageURL, character.ReferenceImages)
}
```

### 3. 视频生成模块 (Video Service)

```go
type VideoService struct {
    klingClient   *kling.Client
    dreaminaClient *dreamina.Client
    animatediff   *animatediff.Client
}

func (s *VideoService) GenerateVideo(ctx context.Context, req GenerateVideoRequest) (*Video, error) {
    // 1. 选择视频生成策略
    provider := s.selectProvider(ctx, req)
    
    // 2. 生成视频
    switch provider {
    case "kling":
        return s.klingClient.Generate(ctx, req.ImageURL, req.Prompt)
    case "dreamina":
        return s.dreaminaClient.Generate(ctx, req.ImageURL, req.Prompt)
    case "animatediff":
        return s.animatediff.Generate(ctx, req.ImageURL)
    }
}
```

### 4. 音频生成模块 (Audio Service)

```go
type AudioService struct {
    gptsovits    *gptsovits.Client
    chattts      *chattts.Client
    yue          *yue.Client
}

func (s *AudioService) GenerateVoice(ctx context.Context, req GenerateVoiceRequest) (*Audio, error) {
    // 使用GPT-SoVITS克隆声音
    if req.VoiceClone {
        return s.gptsovits.Generate(ctx, req.Text, req.ReferenceAudio)
    }
    // 使用ChatTTS生成对话
    return s.chattts.Generate(ctx, req.Text, req.SpeakerID)
}

func (s *AudioService) GenerateMusic(ctx context.Context, req GenerateMusicRequest) (*Audio, error) {
    return s.yue.Generate(ctx, req.Prompt, req.Duration)
}
```
